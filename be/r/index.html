10L + 66L   # 76    # цэлалікавы плюс цэлалікавы дае цэлалікавы

class(NaN)  # "numeric"
# Вы можаце рабіць арыфметычныя аперацыі з двума вектарамі з даўжынёй,

# кратная даўжыні меншага
